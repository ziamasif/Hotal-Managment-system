#include <iostream>
using namespace std;

class HotelManagement
{
private:
    int quant, choice;

    int Qrooms, Qpasta, Qburger, Qshake, Qchicken;

    int Srooms, Spasta, Sburger, Sshake, Schicken;

    int Total_rooms, Total_pasta, Total_burger, Total_shake, Total_chicken;

public:
    HotelManagement()
    {
        Qrooms = Qpasta = Qburger = Qshake = Qchicken = 0;
        Srooms = Spasta = Sburger = Sshake = Schicken = 0;
        Total_rooms = Total_pasta = Total_burger = Total_shake = Total_chicken = 0;
    }

    void setQuantity()
    {
        cout << "\n\t Quantity of items we have:";
        cout << "\n\t Rooms Available: ";
        cin >> Qrooms;
        cout << "\n\t Quantity of Pasta: ";
        cin >> Qpasta;
        cout << "\n\t Quantity of Burger: ";
        cin >> Qburger;
        cout << "\n\t Quantity of Shake: ";
        cin >> Qshake;
        cout << "\n\t Quantity of Chicken Roll: ";
        cin >> Qchicken;
    }

    void menu()
    {
        while (true)
        {
            cout << "\n\n\t\t Please select from the menu options";
            cout << "\n1) Rooms";
            cout << "\n2) Pasta";
            cout << "\n3) Burger";
            cout << "\n4) Shake";
            cout << "\n5) Chicken Roll";
            cout << "\n6) Sales and Collection";
            cout << "\n7) Exit";
            cout << "\n\nEnter your choice: ";
            cin >> choice;

            switch (choice)
            {
            case 1: bookRooms(); break;
            case 2: orderPasta(); break;
            case 3: orderBurger(); break;
            case 4: orderShake(); break;
            case 5: orderChicken(); break;
            case 6: showSales(); break;
            case 7: exit(0);
            default:
                cout << "\nInvalid choice!";
            }
        }
    }

    void bookRooms()
    {
        cout << "\nEnter number of rooms: ";
        cin >> quant;

        if (Qrooms - Srooms >= quant)
        {
            Srooms += quant;
            Total_rooms += quant * 25000;
            cout << "\n" << quant << " Rooms allotted!";
        }
        else
            cout << "\nOnly " << Qrooms - Srooms << " rooms available!";
    }

    void orderPasta()
    {
        cout << "\nEnter pasta quantity: ";
        cin >> quant;

        if (Qpasta - Spasta >= quant)
        {
            Spasta += quant;
            Total_pasta += quant * 250;
            cout << "\n" << quant << " Pasta ordered!";
        }
        else
            cout << "\nOnly " << Qpasta - Spasta << " pasta available!";
    }

    void orderBurger()
    {
        cout << "\nEnter burger quantity: ";
        cin >> quant;

        if (Qburger - Sburger >= quant)
        {
            Sburger += quant;
            Total_burger += quant * 350;
            cout << "\n" << quant << " Burger ordered!";
        }
        else
            cout << "\nOnly " << Qburger - Sburger << " burger available!";
    }

    void orderShake()
    {
        cout << "\nEnter shake quantity: ";
        cin >> quant;

        if (Qshake - Sshake >= quant)
        {
            Sshake += quant;
            Total_shake += quant * 450;
            cout << "\n" << quant << " Shake ordered!";
        }
        else
            cout << "\nOnly " << Qshake - Sshake << " shake available!";
    }

    void orderChicken()
    {
        cout << "\nEnter chicken roll quantity: ";
        cin >> quant;

        if (Qchicken - Schicken >= quant)
        {
            Schicken += quant;
            Total_chicken += quant * 550;
            cout << "\n" << quant << " Chicken roll ordered!";
        }
        else
            cout << "\nOnly " << Qchicken - Schicken << " chicken roll available!";
    }

    void showSales()
    {
        cout << "\n\n========= SALES REPORT =========";
        cout << "\nRooms Sold: " << Srooms << " | Collection: " << Total_rooms;
        cout << "\nPasta Sold: " << Spasta << " | Collection: " << Total_pasta;
        cout << "\nBurger Sold: " << Sburger << " | Collection: " << Total_burger;
        cout << "\nShake Sold: " << Sshake << " | Collection: " << Total_shake;
        cout << "\nChicken Sold: " << Schicken << " | Collection: " << Total_chicken;

        cout << "\n\nTotal Collection: "
             << Total_rooms + Total_pasta + Total_burger + Total_shake + Total_chicken;
    }
};

int main()
{
    HotelManagement h;
    h.setQuantity();
    h.menu();
    return 0;
}
